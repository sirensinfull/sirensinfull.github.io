   INSTALL: Add <script src="ouroboros-pro.js"></script> after ouroboros.js in index.html

   - Progress bar with click-to-seek
   - Now Playing display with track info
   - Google Drive MP3 browser (needs API key)
   - Google Drive Slideshow loader (needs API key)
   
   ARCHITECTURE:
   - Hooks into existing EventBus if found
   - Falls back to standalone Audio element if no engine detected
   - All UI injected dynamically — zero changes to index.html required
   ═══════════════════════════════════════════════════════════════ */

    playlist: [],           // Array of {name, file, objectUrl, source}
    currentIndex: -1,
    isPlaying: false,
    repeatMode: 'off',      // 'off' | 'all' | 'one'
    shuffleOn: false,
    shuffleQueue: [],
    shufflePosition: -1,
    seekInterval: null,
    seekSpeed: 1,
    audio: null,            // The HTML Audio element we control
    engineFound: false,
    driveCache: new Map(),  // folderId → [{id, name, mimeType}]
    driveBlobUrl: null,     // Current blob URL to revoke after play
  };

  // ─── WAIT FOR DOM + EXISTING ENGINE ───────────────────────
  function init() {
    // Try to find the existing AudioEngine
    if (window.OuroborosEngine && window.OuroborosEngine.audio) {
      state.audio = window.OuroborosEngine.audio;
      state.engineFound = true;
    } else {
      // Look for any <audio> element the existing engine created

      if (existingAudio) {

      } else {

        state.audio.crossOrigin = 'anonymous';
      }
    }

    if (window.OuroborosEngine && window.OuroborosEngine.playlist) {
      state.playlist = window.OuroborosEngine.playlist;
    }

    if (window.EventBus) {
      window.EventBus.on('track:load', (data) => {

      });
      window.EventBus.on('track:play', () => { state.isPlaying = true; updateTransportUI(); });
      window.EventBus.on('track:pause', () => { state.isPlaying = false; updateTransportUI(); });
      window.EventBus.on('track:end', () => handleTrackEnd());
      window.EventBus.on('playlist:update', (pl) => {
        if (pl) state.playlist = pl;
        rebuildShuffleQueue();
      });
    }

    // Also listen directly on the audio element
    state.audio.addEventListener('ended', handleTrackEnd);
    state.audio.addEventListener('play', () => { state.isPlaying = true; updateTransportUI(); });
    state.audio.addEventListener('pause', () => { state.isPlaying = false; updateTransportUI(); });
    state.audio.addEventListener('timeupdate', updateProgress);

    injectCSS();
    injectTransportBar();
    injectDrivePanel();

    syncExistingPlaylist();

  // ─── SYNC EXISTING PLAYLIST ───────────────────────────────
  function syncExistingPlaylist() {
    // Watch the playlist panel for changes
    const playlistEl = document.getElementById('playlist-items') || document.getElementById('playlist');
    if (playlistEl) {
      const observer = new MutationObserver(() => {
        const items = playlistEl.querySelectorAll('[data-index], .playlist-item, li');
        if (items.length > 0 && items.length !== state.playlist.length) {
          state.playlist = Array.from(items).map((el, i) => ({
            name: el.textContent.trim(),
            index: i,
            source: 'local'
          }));
          rebuildShuffleQueue();
        }
      });
      observer.observe(playlistEl, { childList: true, subtree: true });
    }
  }

  // ─── TRANSPORT CONTROLS ───────────────────────────────────
  function handleTrackEnd() {
    state.isPlaying = false;
    cleanupDriveBlob();

    if (state.repeatMode === 'one') {
      // Replay current track
      state.audio.currentTime = 0;
      state.audio.play();
      state.isPlaying = true;
    } else if (state.shuffleOn) {
      playNextShuffle();
    } else if (state.repeatMode === 'all') {
      const next = (state.currentIndex + 1) % Math.max(state.playlist.length, 1);
      loadTrack(next);
    } else {
      // Normal: play next, stop at end
      if (state.currentIndex < state.playlist.length - 1) {
        loadTrack(state.currentIndex + 1);
      } else {
        updateTransportUI();
      }
    }
  }

  function loadTrack(index) {
    if (index < 0 || index >= state.playlist.length) return;
    state.currentIndex = index;
    const track = state.playlist[index];

    if (track.objectUrl) {
      state.audio.src = track.objectUrl;
    } else if (track.file) {
      const url = URL.createObjectURL(track.file);
      state.audio.src = url;
    } else if (track.driveId) {
      streamFromDrive(track.driveId, track.name);
      return;
    }

    state.audio.play().catch(() => {});
    state.isPlaying = true;

    if (window.EventBus) {
      window.EventBus.emit('track:load', { index, name: track.name });
    }
    updateNowPlaying();
    updateTransportUI();
  }

  function togglePlayPause() {
    if (!state.audio.src && state.playlist.length > 0) {
      loadTrack(0);
      return;
    }
    if (state.isPlaying) {
      state.audio.pause();
    } else {
      state.audio.play().catch(() => {});
    }
  }

  function skipNext() {
    if (state.shuffleOn) {
      playNextShuffle();
    } else if (state.playlist.length > 0) {
      const next = (state.currentIndex + 1) % state.playlist.length;
      loadTrack(next);
    }
  }

  function skipPrev() {
    // If more than 3 seconds in, restart current track
    if (state.audio.currentTime > 3) {
      state.audio.currentTime = 0;
      return;
    }
    if (state.shuffleOn) {
      playPrevShuffle();
    } else if (state.playlist.length > 0) {
      const prev = (state.currentIndex - 1 + state.playlist.length) % state.playlist.length;
      loadTrack(prev);
    }
  }

  function startSeek(direction) {
    state.seekSpeed = 2;
    state.seekInterval = setInterval(() => {
      state.audio.currentTime = Math.max(0,
        Math.min(state.audio.duration || 0,
          state.audio.currentTime + (direction * state.seekSpeed)));
      state.seekSpeed = Math.min(state.seekSpeed * 1.15, 30); // Accelerate
    }, 100);
  }

  function stopSeek() {
    if (state.seekInterval) {
      clearInterval(state.seekInterval);
      state.seekInterval = null;
      state.seekSpeed = 1;
    }
  }

  function cycleRepeat() {
    const modes = ['off', 'all', 'one'];
    const idx = modes.indexOf(state.repeatMode);
    state.repeatMode = modes[(idx + 1) % 3];
    updateTransportUI();
  }

  function toggleShuffle() {
    state.shuffleOn = !state.shuffleOn;
    if (state.shuffleOn) rebuildShuffleQueue();
    updateTransportUI();
  }

  // ─── SHUFFLE (Fisher-Yates) ───────────────────────────────
  function rebuildShuffleQueue() {
    const indices = state.playlist.map((_, i) => i);
    // Fisher-Yates
    for (let i = indices.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [indices[i], indices[j]] = [indices[j], indices[i]];
    }
    state.shuffleQueue = indices;
    state.shufflePosition = -1;
  }

  function playNextShuffle() {
    if (state.shuffleQueue.length === 0) rebuildShuffleQueue();
    state.shufflePosition++;
    if (state.shufflePosition >= state.shuffleQueue.length) {
      if (state.repeatMode === 'all') {
        rebuildShuffleQueue();
        state.shufflePosition = 0;
      } else {
        state.isPlaying = false;
        updateTransportUI();
        return;
      }
    }
    loadTrack(state.shuffleQueue[state.shufflePosition]);
  }

  function playPrevShuffle() {
    if (state.shufflePosition > 0) {
      state.shufflePosition--;
      loadTrack(state.shuffleQueue[state.shufflePosition]);
    }
  }

  // ─── GOOGLE DRIVE STREAMING ───────────────────────────────
  async function streamFromDrive(fileId, fileName) {
    if (!GDRIVE_API_KEY) {
      showDriveStatus('API key not configured');
      return;
    }

    showDriveStatus(`Loading: ${fileName}...`);
    cleanupDriveBlob();

    try {
      const resp = await fetch(
        `https://www.googleapis.com/drive/v3/files/${fileId}?alt=media&key=${GDRIVE_API_KEY}`
      );
      if (!resp.ok) throw new Error(`Drive API ${resp.status}`);
      const blob = await resp.blob();
      state.driveBlobUrl = URL.createObjectURL(blob);
      state.audio.src = state.driveBlobUrl;
      await state.audio.play();
      state.isPlaying = true;
      showDriveStatus(`Playing: ${fileName}`);
      updateNowPlaying();
      updateTransportUI();
    } catch (err) {
      showDriveStatus(`Error: ${err.message}`);
      console.error('[OuroborosPro] Drive stream error:', err);
    }
  }

  function cleanupDriveBlob() {
    if (state.driveBlobUrl) {
      URL.revokeObjectURL(state.driveBlobUrl);
      state.driveBlobUrl = null;
    }
  }

  async function browseDriveFolder(folderId, breadcrumbs) {
    if (!GDRIVE_API_KEY) {
      showDriveStatus('Enter your Google Drive API key in ouroboros-pro.js');
      return;
    }

    // Check cache
    if (state.driveCache.has(folderId)) {
      renderDriveContents(state.driveCache.get(folderId), breadcrumbs);
      return;
    }

    showDriveStatus('Loading folder...');
    try {
      const q = encodeURIComponent(`'${folderId}' in parents and trashed=false`);
      const fields = encodeURIComponent('files(id,name,mimeType,size)');
      const resp = await fetch(
        `https://www.googleapis.com/drive/v3/files?q=${q}&fields=${fields}&orderBy=name&pageSize=200&key=${GDRIVE_API_KEY}`
      );
      if (!resp.ok) throw new Error(`API ${resp.status}`);
      const data = await resp.json();
      const files = data.files || [];

      // Sort: folders first, then files
      files.sort((a, b) => {
        const aFolder = a.mimeType === 'application/vnd.google-apps.folder' ? 0 : 1;
        const bFolder = b.mimeType === 'application/vnd.google-apps.folder' ? 0 : 1;
        if (aFolder !== bFolder) return aFolder - bFolder;
        return a.name.localeCompare(b.name);
      });

      state.driveCache.set(folderId, files);

    } catch (err) {
      showDriveStatus(`Error: ${err.message}`);
    }
  }

  function renderDriveContents(files, breadcrumbs) {
    const list = document.getElementById('opro-drive-list');
    if (!list) return;
    list.innerHTML = '';

    // Breadcrumb nav
    const breadEl = document.createElement('div');
    breadEl.className = 'opro-breadcrumbs';
    breadcrumbs.forEach((crumb, i) => {
      const span = document.createElement('span');
      span.textContent = crumb.name;
      span.className = 'opro-crumb';
      if (i < breadcrumbs.length - 1) {
        span.style.cursor = 'pointer';
        span.style.opacity = '0.6';
        span.addEventListener('click', () => {
          browseDriveFolder(crumb.id, breadcrumbs.slice(0, i + 1));
        });
      }
      breadEl.appendChild(span);
      if (i < breadcrumbs.length - 1) {
        const sep = document.createElement('span');
        sep.textContent = ' › ';
        sep.style.opacity = '0.3';
        breadEl.appendChild(sep);
      }
    });
    list.appendChild(breadEl);

    // Select All MP3s button
    const mp3Files = files.filter(f => f.name.toLowerCase().endsWith('.mp3'));
    if (mp3Files.length > 0) {
      const selectAll = document.createElement('div');
      selectAll.className = 'opro-drive-item opro-select-all';
      selectAll.innerHTML = `<span class="opro-icon">⊕</span> Queue All ${mp3Files.length} MP3s`;
      selectAll.addEventListener('click', () => {
        mp3Files.forEach(f => addDriveTrackToPlaylist(f.id, f.name));
        showDriveStatus(`Queued ${mp3Files.length} tracks`);
      });
      list.appendChild(selectAll);
    }

    // Image files for slideshow
    const imageFiles = files.filter(f =>
      /\.(jpg|jpeg|png|gif|webp|bmp)$/i.test(f.name));
    if (imageFiles.length > 0) {
      const loadSlides = document.createElement('div');
      loadSlides.className = 'opro-drive-item opro-select-all opro-slide-btn';
      loadSlides.innerHTML = `<span class="opro-icon">🖼</span> Load ${imageFiles.length} Images → Slideshow`;
      loadSlides.addEventListener('click', () => {
        loadDriveSlideshowImages(imageFiles);
      });
      list.appendChild(loadSlides);
    }

    // File list
    files.forEach(f => {
      const isFolder = f.mimeType === 'application/vnd.google-apps.folder';
      const isAudio = f.name.toLowerCase().endsWith('.mp3');
      const isImage = /\.(jpg|jpeg|png|gif|webp|bmp)$/i.test(f.name);

      const item = document.createElement('div');
      item.className = 'opro-drive-item';
      if (!isFolder && !isAudio && !isImage) item.style.opacity = '0.35';

      const icon = isFolder ? '📁' : isAudio ? '♫' : isImage ? '🖼' : '📄';
      item.innerHTML = `<span class="opro-icon">${icon}</span><span class="opro-fname">${f.name}</span>`;

      if (isFolder) {
        item.addEventListener('click', () => {
          const newCrumbs = [...breadcrumbs, { id: f.id, name: f.name }];
          browseDriveFolder(f.id, newCrumbs);
        });
      } else if (isAudio) {
        item.addEventListener('click', () => {
          addDriveTrackToPlaylist(f.id, f.name);
          showDriveStatus(`Queued: ${f.name}`);
        });
      } else if (isImage) {
        item.addEventListener('click', () => {
          loadSingleDriveImage(f.id, f.name);
        });
      }

      list.appendChild(item);
    });
  }

  function addDriveTrackToPlaylist(driveId, name) {
    const track = { name: name.replace(/\.mp3$/i, ''), driveId, source: 'drive' };
    state.playlist.push(track);

    // If nothing playing, start this track
    if (state.currentIndex === -1 || !state.isPlaying) {
      loadTrack(state.playlist.length - 1);
    }

    rebuildShuffleQueue();
    updatePlaylistCount();
  }

  // ─── GOOGLE DRIVE SLIDESHOW ───────────────────────────────
  async function loadDriveSlideshowImages(imageFiles) {
    if (!GDRIVE_API_KEY) return;

    showDriveStatus(`Loading ${imageFiles.length} slideshow images...`);

    for (const img of imageFiles) {
      try {
        const resp = await fetch(
          `https://www.googleapis.com/drive/v3/files/${img.id}?alt=media&key=${GDRIVE_API_KEY}`
        );
        if (!resp.ok) continue;
        const blob = await resp.blob();
        const url = URL.createObjectURL(blob);

        // Push to SlideshowEngine if it exists
        if (window.OuroborosEngine && window.OuroborosEngine.slideshow) {
          window.OuroborosEngine.slideshow.addImage(url);
        } else if (window.EventBus) {
          window.EventBus.emit('slideshow:add', { url, name: img.name });
        }

        // Also try the direct slideshow approach
        addToSlideshow(url, img.name);
      } catch (e) {
        console.warn('[OuroborosPro] Failed to load image:', img.name, e);
      }
    }
    showDriveStatus(`Loaded ${imageFiles.length} images to slideshow`);
  }

  async function loadSingleDriveImage(fileId, name) {
    if (!GDRIVE_API_KEY) return;
    showDriveStatus(`Loading: ${name}...`);
    try {
      const resp = await fetch(
        `https://www.googleapis.com/drive/v3/files/${fileId}?alt=media&key=${GDRIVE_API_KEY}`
      );
      const blob = await resp.blob();
      const url = URL.createObjectURL(blob);
      addToSlideshow(url, name);
      showDriveStatus(`Added to slideshow: ${name}`);
    } catch (e) {
      showDriveStatus(`Error: ${e.message}`);
    }
  }

  function addToSlideshow(url, name) {
    // Try multiple integration points
    if (window.EventBus) {
      window.EventBus.emit('slideshow:add', { url, name });
    }
    // Direct DOM: look for slideshow image container
    const container = document.getElementById('slideshow-images') ||
                      document.getElementById('slideshow-container') ||
                      document.querySelector('.slideshow');
    if (container) {
      const img = document.createElement('img');
      img.src = url;
      img.alt = name;
      img.style.display = 'none';
      container.appendChild(img);
    }
  }

      timeEl.textContent = `${fmtTime(state.audio.currentTime)} / ${fmtTime(state.audio.duration)}`;
    }
  }

  function fmtTime(s) {
    if (!s || isNaN(s)) return '0:00';
    const m = Math.floor(s / 60);
    const sec = Math.floor(s % 60);
    return `${m}:${sec.toString().padStart(2, '0')}`;
  }
